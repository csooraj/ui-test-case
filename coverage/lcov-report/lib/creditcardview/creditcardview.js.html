/**
 * Credit Card View Component
 * author: soorajc
 * created@: 25/07/2018
 */
 
import React, { Component } from 'react';
import {
  View,
  ImageBackground,
  ViewPropTypes,
  Image,
  Text,
} from 'react-native';
import PropTypes from 'prop-types';
 
import { MARGIN_OFFSET, MARGIN_VALUE } from './config';
import Styles from './styles';
 
const creditCardImage = require('../../assets/creditcard.png');
 
export default class CreditCardView extends Component<> {
 
  render() {
    const cardHeight = (this.props.name.length / MARGIN_OFFSET) + MARGIN_VALUE;
    const cardNumbers = this.props.cardNumber.split('-');
    return (
      <View
        style={[
          this.props.cardContainerStyle
          ?
          this.props.cardContainerStyle
          :
          Styles.container,
          { height: cardHeight + '%',
          }]
        }
      >
        <ImageBackground
          resizeMode="stretch"
          source={this.props.creditCardImage}
          style={
            this.props.creditCardImageStyle
            ?
            this.props.creditCardImageStyle
            :
            Styles.creditCardImage
          }
        >
          <View
            style={
              this.props.cardNumberContainerStyle
              ?
              this.props.cardNumberContainerStyle
              :
              Styles.cardNumberContainer
             }
          >
            <Text
              style={
              [Styles.cardNumber,
                { color: this.props.cardLabelColor,
                },
              ]
             }
            >
              {cardNumbers[0]} {' '} {cardNumbers[1]} {' '} {cardNumbers[2]} { ' ' } {cardNumbers[3]}
            </Text>
            <View
              style={
                this.props.additionDataContainerStyle
                ?
                this.props.additionDataContainerStyle
                :
                Styles.additionalDataView
              }
            >
              <Text
                style={
                [this.props.cvvStyle ? this.props.cvvStyle : Styles.cvv,
                  { color: this.props.cardLabelColor,
                  },
                ]
               }
              >
                {this.props.cvv}
              </Text>
              <Text
                style={
                [this.props.validLabelStyle ? this.props.validLabelStyle : Styles.validLabel,
                  { color: this.props.cardLabelColor,
                  },
                ]
                }
              >
                VALID {'\n'} THRU
              </Text>
              <Text
                style={
                [this.props.dateStyle ? this.props.dateStyle : Styles.date,
                  { color: this.props.cardLabelColor,
                  },
                ]
                }
              >
                {this.props.expiryDate}
              </Text>
            </View>
            <Text
              style={
              [this.props.cardHolderNameStyle
                ?
                this.props.cardHolderNameStyle : Styles.cardHolderName,
                { color: this.props.cardLabelColor },
              ]
              }
            >
              {this.props.name}
            </Text>
          </View>
        </ImageBackground>
      </View>
    );
  }
}
 
CreditCardView.propTypes = {
  additionDataContainerStyle: ViewPropTypes.style,
  cardContainerStyle: ViewPropTypes.style,
  cardHolderNameStyle: Text.propTypes.style,
  cardLabelColor: PropTypes.string,
  cardNumber: PropTypes.string,
  cardNumberContainerStyle: ViewPropTypes.style,
  creditCardImage: Image.propTypes.source,
  creditCardImageStyle: Image.propTypes.style,
  cvv: PropTypes.string,
  cvvStyle: Text.propTypes.style,
  dateStyle: Text.propTypes.style,
  expiryDate: PropTypes.string,
  name: PropTypes.string,
  validLabelStyle: Text.propTypes.style,
};
 
CreditCardView.defaultProps = {
  additionDataContainerStyle: '',
  cardContainerStyle: '',
  cardHolderNameStyle: '',
  cardLabelColor: '#1A2534',
  cardNumber: '',
  cardNumberContainerStyle: '',
  creditCardImage: creditCardImage,
  creditCardImageStyle: '',
  cvv: '',
  cvvStyle: '',
  dateStyle: '',
  expiryDate: '',
  validLabelStyle: '',
  name: '',
};
