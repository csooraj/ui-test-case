/**
 * TextInputFieldWithIcon Component
 * author: soorajc
 * created@: 18/07/2018
 */
 
import React, { Component } from 'react';
import {
  View,
  TextInput,
  TouchableOpacity,
  ViewPropTypes,
  Text,
} from 'react-native';
import PropTypes from 'prop-types';
import Icon from 'react-native-vector-icons/FontAwesome';
 
import Styles from './styles';
 
export default class TextInputFieldWithIcon extends Component<> {
 
  constructor(props) {
    super(props);
    this.state = {
      text: '',
      error: false,
    };
  }
 
  formValidator(value) {
    this.setState({ text: value });
    const reg = this.props.regex;
    const placeholder = this.props.placeholder;
    if (reg.test(value)) {
      this.setState({ error: false });
      this.props.formErrorHandler(false, placeholder);
    } else {
      this.setState({ error: true });
      this.props.formErrorHandler(true, placeholder);
    }
  }
 
  render() {
    const inputContainerStyle = this.props.inputContainerStyle
    ?
    this.props.inputContainerStyle
    :
    Styles.inputContainer;
 
    const errorContainerStyle = this.props.errorContainerStyle
    ?
    this.props.errorContainerStyle
    :
    Styles.errorContainer;
 
    return (
      <View>
        {
          this.state.error
          ?
            <Text
              adjustsFontSizeToFit
              style={
                this.props.labelErrorStyle
                ?
                this.props.labelErrorStyle
                :
                Styles.errorLabel
              }
            >
              {this.props.errorMessage}
            </Text>
          :
            <Text />
        }
        <View style={this.state.error ? errorContainerStyle : inputContainerStyle}>
          {
            this.props.iconOrientation === 'left'
            ?
              <View style={Styles.iconContainer}>
                <Icon
                  name={this.props.iconName}
                  size={25}
                  color={this.props.iconColor}
                  style={Styles.icon}
                />
              </View>
            :
              <View />
          }
          <TextInput
            ref={this.props.textInputRef}
            testID={'InputBox'}
            placeholder={this.props.placeholder}
            placeholderTextColor={this.props.placeholderTextColor}
            secureTextEntry={this.props.secureTextEntry}
            underlineColorAndroid={this.props.underlineColorAndroid}
            style={this.props.textInputStyle ? this.props.textInputStyle : Styles.defaultInput}
            onChangeText={text => this.formValidator(text)}
            value={this.state.text}
          />
          {
            this.props.iconOrientation === 'right'
            ?
              <TouchableOpacity
                testID={'IconButton'}
                onPress={() => this.props.buttonPressHandler()}
                style={
                  this.props.iconContainerStyle
                  ?
                  this.props.iconContainerStyle
                  :
                  Styles.iconContainer
                }
              >
                {
                  this.props.showButtonText
                  ?
                    <Text style={Styles.buttonLabel}>
                      {this.props.buttonLabel}
                    </Text>
                  :
                    <Icon
                      name={this.props.iconName}
                      size={this.props.iconSize}
                      color={this.props.iconColor}
                      style={Styles.icon}
                    />
                }
              </TouchableOpacity>
            :
              <View />
          }
        </View>
      </View>
    );
  }
}
 
 
TextInputFieldWithIcon.propTypes = {
  buttonLabel: PropTypes.string,
  buttonPressHandler: PropTypes.func,
  errorContainerStyle: ViewPropTypes.style,
  errorMessage: PropTypes.string,
  formErrorHandler: PropTypes.func,
  iconColor: PropTypes.string,
  iconContainerStyle: ViewPropTypes.style,
  iconName: PropTypes.string,
  iconOrientation: PropTypes.string,
  iconSize: PropTypes.number,
  inputContainerStyle: ViewPropTypes.style,
  labelErrorStyle: TextInput.propTypes.style,
  placeholder: TextInput.propTypes.placeholder,
  placeholderTextColor: TextInput.propTypes.placeholderTextColor,
  regex: PropTypes.shape(),
  secureTextEntry: TextInput.propTypes.secureTextEntry,
  showButtonText: PropTypes.bool,
  textInputRef: PropTypes.oneOfType([PropTypes.func, PropTypes.object]),
  textInputStyle: TextInput.propTypes.style,
  underlineColorAndroid: TextInput.propTypes.underlineColorAndroid,
};
 
TextInputFieldWithIcon.defaultProps = {
  buttonLabel: '',
  buttonPressHandler: '',
  errorContainerStyle: '',
  errorMessage: '',
  formErrorHandler: '',
  iconColor: '',
  iconContainerStyle: '',
  iconName: '',
  iconOrientation: '',
  iconOrientationLeft: false,
  iconOrientationRight: false,
  iconSize: 25,
  inputContainerStyle: '',
  labelErrorStyle: '',
  placeholder: '',
  placeholderTextColor: '',
  regex: '',
  secureTextEntry: false,
  showButtonText: false,
  textInputRef: '',
  textInputStyle: '',
  textInputErrorStyle: '',
  underlineColorAndroid: '',
};
