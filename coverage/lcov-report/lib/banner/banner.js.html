 * Banner Component
 * author: soorajc
 * created@: 23/07/2018
 */
 
import React, { Component } from 'react';
import PropTypes from 'prop-types';
import {
 View,
 TouchableOpacity,
 Image,
 ViewPropTypes,
 ImageBackground,
 Text,
} from 'react-native';
 
import { NUMBER_OF_TEXT_LINE } from './config';
import Styles from './styles';
 
export default class Banner extends Component<> {
 
  render() {
    return (
      <TouchableOpacity
        style={
          this.props.bannerContainerStyle
          ?
          this.props.bannerContainerStyle
          :
          Styles.container
        }
        testID={'BannerButton'}
        onPress={() => this.props.bannerPressHandler(this.props.bannerNo)}
      >
        <ImageBackground
          source={this.props.bannerImage}
          resizeMode="cover"
          style={
            this.props.imageBackgroundStyle
            ?
            this.props.imageBackgroundStyle
            :
            Styles.imageBackground
          }
        >
          <View
            style={
            this.props.bannerTitleContainerStyle
            ?
            this.props.bannerTitleContainerStyle
            :
            Styles.bannerTitleContainer
          }
          >
            <Text
              numberOfLines={NUMBER_OF_TEXT_LINE}
              style={
                this.props.bannerTitleStyle
                ?
                [this.props.bannerTitleStyle, { color: this.props.titleColor }]
                :
                [Styles.bannerTitle, { color: this.props.titleColor }]
              }
            >
              {this.props.bannerTitle}
            </Text>
          </View>
        </ImageBackground>
      </TouchableOpacity>
    );
  }
}
 
 
Banner.propTypes = {
  bannerContainerStyle: ViewPropTypes.style,
  bannerImage: Image.propTypes.source,
  bannerNo: PropTypes.number,
  bannerPressHandler: PropTypes.func,
  bannerTitle: PropTypes.string,
  bannerTitleContainerStyle: ViewPropTypes.style,
  bannerTitleStyle: Text.propTypes.style,
  imageBackgroundStyle: Image.propTypes.style,
  titleColor: PropTypes.string,
};
 
Banner.defaultProps = {
  bannerContainerStyle: '',
  bannerImage: '',
  bannerNo: '',
  bannerPressHandler: '',
  bannerTitle: '',
  bannerTitleContainerStyle: '',
  bannerTitleStyle: '',
  imageBackgroundStyle: '',
  titleColor: 'black',
};
