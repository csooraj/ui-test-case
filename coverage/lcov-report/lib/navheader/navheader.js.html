 * author: soorajc
 * created@: 27/07/2018

import React, { Component } from 'react';
import {
  View,
  ViewPropTypes,
  TouchableOpacity,
} from 'react-native';
import PropTypes from 'prop-types';
import Icon from 'react-native-vector-icons/FontAwesome';
 
import Styles from './styles';
 
 
export default class NavHeader extends Component<> {
  render() {
    return (
      <View
        style={
          this.props.containerStyle
          ?
          this.props.containerStyle
          :
          Styles.container
        }
      >
        <TouchableOpacity
          testID={'FrontIconButton'}
          onPress={() => this.props.onPressFrontIcon()}
          style={
            this.props.frontIconContainerStyle
            ?
            this.props.frontIconContainerStyle
            :
            Styles.frontIconContainer
          }
        >
          <Icon
            name={this.props.frontIcon}
            size={this.props.frontIconSize}
            color={this.props.frontIconColor}
          />
        </TouchableOpacity>
        <View
          style={
            this.props.middleContainerStyle
            ?
            this.props.middleContainerStyle
            :
            Styles.middleContainer
          }
        >
          {this.props.middleComponent}
        </View>
        {
          this.props.endIcon
          ?
            <TouchableOpacity
              testID={'EndIconButton'}
              onPress={() => this.props.onPressEndIcon()}
              style={
                this.props.endIconContainerStyle
                ?
                this.props.endIconContainerStyle
                :
                Styles.endIconContainer
              }
            >
              <Icon
                name={this.props.endIcon}
                size={this.props.endIconSize}
                color={this.props.endIconColor}
              />
            </TouchableOpacity>
            :
            <View />
        }
      </View>
    );
  }
}
 
NavHeader.propTypes = {
  containerStyle: ViewPropTypes.style,
  endIcon: PropTypes.string,
  endIconColor: PropTypes.string,
  endIconContainerStyle: ViewPropTypes.style,
  endIconSize: PropTypes.number,
  frontIcon: PropTypes.string,
  frontIconColor: PropTypes.string,
  frontIconContainerStyle: ViewPropTypes.style,
  frontIconSize: PropTypes.number,
  middleComponent: PropTypes.node,
  middleContainerStyle: ViewPropTypes.style,
  onPressEndIcon: PropTypes.func,
  onPressFrontIcon: PropTypes.func,
};
 
NavHeader.defaultProps = {
  containerStyle: '',
  endIcon: '',
  endIconColor: 'black',
  endIconContainerStyle: '',
  endIconSize: 30,
  frontIcon: 'bars',
  frontIconColor: 'black',
  frontIconContainerStyle: '',
  frontIconSize: 30,
  middleComponent: '',
  middleContainerStyle: '',
  onPressFrontIcon: null,
  onPressEndIcon: null,
};
