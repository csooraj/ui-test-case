/**
 * Collapsible Component
 * author: soorajc
 * created@: 25/07/2018
 */
 
import React, { Component } from 'react';
import {
  View,
  Text,
  TouchableOpacity,
  ViewPropTypes,
} from 'react-native';
import PropTypes from 'prop-types';
 
import Icon from 'react-native-vector-icons/FontAwesome';
import Styles from './styles';
 
export default class Collapsible extends Component<> {
 
  constructor(props) {
    super(props);
    this.state = {
      collapsed: false,
    };
  }
 
  handleCollapse() {
    if (this.state.collapsed) {
      this.setState({ collapsed: false });
    } else {
      this.setState({ collapsed: true });
    }
  }
 
 
  render() {
    return (
      <View
        style={
          this.props.containerStyle
          ?
          this.props.containerStyle
          :
          Styles.container
        }
      >
        <TouchableOpacity
          testID={'Collapsible'}
          onPress={() => this.handleCollapse()}
          style={
            this.props.headerContainerStyle
            ?
            this.props.headerContainerStyle
            :
            Styles.headerContainer
          }
        >
          <View
            style={
              this.props.titleContainerStyle
              ?
              this.props.titleContainerStyle
              :
              Styles.titleContainer
            }
          >
            <Text
              style={
                this.props.titleStyle
                ?
                this.props.titleStyle
                :
                Styles.title
              }
            >
              {this.props.title}
            </Text>
          </View>
          <View
            style={
              this.props.iconContainerStyle
              ?
              this.props.iconContainerStyle
              :
              Styles.iconContainer
            }
          >
            <Icon
              name={
                this.state.collapsed
                ?
                this.props.iconCollapsed
                :
                this.props.iconNormal
              }
              size={this.props.iconSize}
              color={
                this.state.collapsed
                ?
                this.props.iconColorCollapsed
                :
                this.props.iconColorNormal
              }
            />
          </View>
        </TouchableOpacity>
        <View style={this.state.collapsed ? Styles.collapsedView : Styles.hideView}>
          {this.props.children}
        </View>
      </View>
    );
  }
}
 
Collapsible.propTypes = {
  children: PropTypes.node,
  containerStyle: ViewPropTypes.style,
  headerContainerStyle: ViewPropTypes.style,
  iconCollapsed: PropTypes.string,
  iconColorCollapsed: PropTypes.string,
  iconColorNormal: PropTypes.string,
  iconContainerStyle: ViewPropTypes.style,
  iconNormal: PropTypes.string,
  iconSize: PropTypes.number,
  title: PropTypes.string,
  titleContainerStyle: ViewPropTypes.style,
  titleStyle: Text.propTypes.style,
};
 
Collapsible.defaultProps = {
  children: '',
  containerStyle: '',
  headerContainerStyle: '',
  iconCollapsed: 'caret-up',
  iconColorCollapsed: 'red',
  iconColorNormal: 'black',
  iconContainerStyle: '',
  iconNormal: 'caret-down',
  iconSize: 35,
  title: '',
  titleContainerStyle: '',
  titleStyle: '',
};
