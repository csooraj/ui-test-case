/**
 * Check Box Component
 * author: soorajc
 * created@: 20/07/2018
 */
 
import React, { Component } from 'react';
import {
  TouchableOpacity,
  View,
  ViewPropTypes,
  Text,
} from 'react-native';
import PropTypes from 'prop-types';
import Icon from 'react-native-vector-icons/MaterialCommunityIcons';
 
import Styles from './styles';
 
export default class CheckBox extends Component<> {
 
  constructor(props) {
    super(props);
    this.state = {
      selected: false,
    };
  }
 
  handleSelect(label) {
    this.props.onSelect(label);
    if (this.state.selected) {
      this.setState({ selected: false });
    } else {
      this.setState({ selected: true });
    }
  }
 
  render() {
    return (
      <TouchableOpacity
        testID={'CheckBox'}
        style={
          this.props.checkBoxContainerStyle
          ?
          this.props.checkBoxContainerStyle
          :
          Styles.checkBoxContainer
        }
        onPress={() => this.handleSelect(this.props.label)}
      >
        <Icon
          name={this.state.selected ? this.props.activeIcon : this.props.passiveIcon}
          size={this.props.size}
          color={this.props.color}
        />
        <View
          style={
            this.props.textContainerStyle
            ?
            this.props.textContainerStyle
            :
            Styles.textContainer
          }
        >
          <Text
            style={
              this.props.labelStyle
              ?
              this.props.labelStyle
              :
              Styles.label
            }
          >
            {this.props.label}
          </Text>
        </View>
      </TouchableOpacity>
    );
  }
}
 
 
CheckBox.propTypes = {
  activeIcon: PropTypes.string,
  checkBoxContainerStyle: ViewPropTypes.style,
  color: PropTypes.string,
  label: PropTypes.string,
  labelStyle: Text.propTypes.style,
  onSelect: PropTypes.func,
  passiveIcon: PropTypes.string,
  size: PropTypes.number,
  textContainerStyle: Text.propTypes.style,
};
 
CheckBox.defaultProps = {
  activeIcon: 'checkbox-marked',
  checkBoxContainerStyle: '',
  color: 'black',
  label: '',
  labelStyle: '',
  onSelect: '',
  passiveIcon: 'square-outline',
  size: 35,
  textContainerStyle: '',
};
