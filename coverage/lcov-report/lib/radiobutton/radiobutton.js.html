/**
 * Radio Button Group Component
 * author: soorajc
 * created@: 20/07/2018
 */
 
import React, { Component } from 'react';
import {
  TouchableOpacity,
  ViewPropTypes,
  View,
  Text,
} from 'react-native';
import PropTypes from 'prop-types';
import Icon from 'react-native-vector-icons/MaterialCommunityIcons';
 
import Styles from './styles';
 
export default class RadioButton extends Component<> {
 
  constructor(props) {
    super(props);
    this.state = {
      currentSelected: this.props.defaultSelectedIndex,
    };
  }
 
  handleChange(index, name) {
    this.setState({ currentSelected: index });
    this.props.onChange(index, name);
  }
 
  render() {
    return (
      <View
        style={
        this.props.radioGroupContainerStyle
        ?
        this.props.radioGroupContainerStyle
        :
        Styles.radioGroupContainer
      }
      >
        {
           this.props.radioGroupItems.map(item =>
             <View
               key={item.no}
               style={
                 this.props.radioContainerStyle
                 ?
                 this.props.radioContainerStyle
                 :
                 Styles.container
               }
             >
               <TouchableOpacity
                 testID={item.name}
                 onPress={() => this.handleChange(item.no, item.name)}
               >
                 <Icon
                   name={this.props.radioIcon}
                   size={this.props.size}
                   color={
                     this.state.currentSelected === item.no
                     ?
                     this.props.activeColor
                     :
                     this.props.normalColor
                  }
                 />
               </TouchableOpacity>
               <View
                 style={
                   this.props.textContainerStyle
                   ?
                   this.props.textContainerStyle
                   :
                   Styles.textContainer
                 }
               >
                 <Text
                   style={
                     this.props.radioLabelStyle
                     ?
                     this.props.radioLabelStyle
                     :
                     Styles.radioLabel
                   }
                 >
                   {item.name}
                 </Text>
               </View>
             </View>,
          )
        }
      </View>
    );
  }
}
 
RadioButton.propTypes = {
  activeColor: PropTypes.string,
  defaultSelectedIndex: PropTypes.number,
  normalColor: PropTypes.string,
  onChange: PropTypes.func,
  radioContainerStyle: ViewPropTypes.style,
  radioGroupContainerStyle: ViewPropTypes.style,
  radioGroupItems: PropTypes.arrayOf(PropTypes.object),
  radioIcon: PropTypes.string,
  radioLabelStyle: Text.propTypes.style,
  size: PropTypes.number,
  textContainerStyle: ViewPropTypes.style,
};
 
RadioButton.defaultProps = {
  activeColor: '#1e88e5',
  defaultSelectedIndex: 1,
  normalColor: '#90caf9',
  onChange: '',
  radioContainerStyle: '',
  radioGroupContainerStyle: '',
  radioGroupItems: '',
  radioIcon: 'radiobox-marked',
  radioLabelStyle: '',
  size: 35,
  textContainerStyle: '',
};
