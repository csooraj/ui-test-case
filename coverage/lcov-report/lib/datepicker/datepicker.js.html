/**
 * DatePicker Component
 * author: soorajc
 * created@: 31/07/2018
 */
 
 
import React, { Component } from 'react';
import {
  View,
  TouchableOpacity,
  ViewPropTypes,
  Text,
} from 'react-native';
 
import PropTypes from 'prop-types';
import DateTimePicker from 'react-native-modal-datetime-picker';
import Icon from 'react-native-vector-icons/FontAwesome';
import DATE_LENGTH from './config';
import Button from '../../lib/button/button';
import Styles from './styles';
 
 
export default class DatePicker extends Component<{}> {
 
  static navigationOptions = {
    header: null,
  }
 
  constructor(props) {
    super(props);
    this.state = {
      isDateTimePickerVisible: false,
      selectedValue: this.props.datePickerLabel,
    };
  }
 
  handleDatePicker() {
    if (this.state.isDateTimePickerVisible) {
      this.setState({ isDateTimePickerVisible: false });
    } else {
      this.setState({ isDateTimePickerVisible: true });
    }
  }
 
  hideDateTimePicker() {
    this.setState({ isDateTimePickerVisible: false });
  }
 
  handleDatePicked = (date) => {
    this.props.onSelect(date);
    const dateString = date.toString();
    const splitDate = dateString.substr(0, DATE_LENGTH);
    this.setState({ selectedValue: splitDate });
    this.hideDateTimePicker();
  };
 
 
  render() {
    return (
      <View
        style={
          this.props.containerStyle
          ?
          this.props.containerStyle
          :
          Styles.container
        }
      >
        <Button
          testID={'PickerButton'}
          label={this.state.selectedValue}
          style={
            this.props.datePickerButtonStyle
            ?
            this.props.datePickerButtonStyle
            :
            Styles.datePickerButton
          }
          labelStyle={
            this.props.datePickerLabelStyle
            ?
            this.props.datePickerLabelStyle
            :
            Styles.datePickerLabel
          }
          onPressHandler={() => this.handleDatePicker()}
        />
        <TouchableOpacity
          testID={'CalendarButton'}
          onPress={() => this.handleDatePicker()}
          style={
            this.props.datePickerIconStyle
            ?
            this.props.datePickerIconStyle
            :
            Styles.datePickerIcon
          }
        >
          <Icon
            name="calendar"
            size={this.props.iconSize}
            color={this.props.iconColor}
          />
        </TouchableOpacity>
        <DateTimePicker
          isVisible={this.state.isDateTimePickerVisible}
          onConfirm={this.handleDatePicked}
          onCancel={() => this.hideDateTimePicker()}
        />
      </View>
    );
  }
}
 
DatePicker.propTypes = {
  containerStyle: ViewPropTypes.style,
  datePickerButtonStyle: ViewPropTypes.style,
  datePickerIconStyle: ViewPropTypes.style,
  datePickerLabel: PropTypes.string,
  datePickerLabelStyle: Text.propTypes.style,
  iconSize: PropTypes.number,
  iconColor: PropTypes.string,
  onSelect: PropTypes.func,
};
 
DatePicker.defaultProps = {
  containerStyle: '',
  datePickerButtonStyle: '',
  datePickerIconStyle: '',
  datePickerLabel: '',
  datePickerLabelStyle: '',
  iconSize: 20,
  iconColor: '#AFB8BB',
  onSelect: null,
};
