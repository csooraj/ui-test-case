/**
 * TabBar Component
 * author: soorajc
 * created@: 19/07/2018
 */
 
import React, { Component } from 'react';
import {
 View,
 TouchableOpacity,
 ViewPropTypes,
 Dimensions,
} from 'react-native';
import PropTypes from 'prop-types';
import Icon from 'react-native-vector-icons/FontAwesome';
 
import Styles from './styles';
 
const { width } = Dimensions.get('window');
 
export default class TabBar extends Component<> {
 
  constructor(props) {
    super(props);
    this.state = {
      currentTab: this.props.defaultSelectedTab,
    };
  }
 
  handleTabBarPress(item) {
    this.props.onPress(item);
    this.setState({ currentTab: item });
  }
 
  render() {
    const tabItemsCount = this.props.tabItems.length;
    return (
      <View
        style={
          this.props.style
          ?
          this.props.style
          :
          Styles.container
        }
      >
        {
           this.props.tabItems.map(item =>
             <TouchableOpacity
               key={item.no}
               testID={'TabButton'}
               onPress={() => this.handleTabBarPress(item.no)}
               style={
                 this.props.tabBarItemActiveStyle
                 &&
                 this.props.tabBarItemPassiveStyle
                 ?
                 [this.state.currentTab === item.no ?
                   this.props.tabBarItemActiveStyle
                   :
                   this.props.tabBarItemPassiveStyle,
                   { width: width / tabItemsCount }]
                 :
                 [this.state.currentTab === item.no
                   ?
                   Styles.activeItem
                   :
                   Styles.passiveItem,
                   { width: width / tabItemsCount }]
               }
             >
               <Icon
                 name={item.name}
                 size={this.props.iconSize}
                 color={
                   this.state.currentTab === item.no
                   ?
                   this.props.activeColor
                   :
                   this.props.normalColor
                 }
               />
             </TouchableOpacity>,
           )
        }
      </View>
    );
  }
}
 
 
TabBar.propTypes = {
  activeColor: PropTypes.string,
  defaultSelectedTab: PropTypes.number,
  iconSize: PropTypes.number,
  normalColor: PropTypes.string,
  onPress: TouchableOpacity.propTypes.onPress,
  style: ViewPropTypes.style,
  tabItems: PropTypes.arrayOf(PropTypes.object),
  tabBarItemActiveStyle: ViewPropTypes.style,
  tabBarItemPassiveStyle: ViewPropTypes.style,
};
 
TabBar.defaultProps = {
  activeColor: 'red',
  defaultSelectedTab: 1,
  iconSize: 14,
  normalColor: 'black',
  onPress: '',
  style: '',
  tabItems: '',
  tabBarItemActiveStyle: '',
  tabBarItemPassiveStyle: '',
};
