/**
 * Notification Bar Component
 * author: soorajc
 * created@: 27/07/2018
 */
 
import React, { Component } from 'react';
import {
  View,
  ViewPropTypes,
  Text,
} from 'react-native';
import PropTypes from 'prop-types';
import Icon from 'react-native-vector-icons/FontAwesome';
 
import Styles from './styles';
 
 
export default class NotificationBar extends Component<> {
  render() {
    return (
      <View
        style={
        [this.props.containerStyle
          ?
          this.props.containerStyle
          :
          Styles.container,
          { backgroundColor: this.props.backgroundColor },
        ]}
      >
        <View
          style={
            this.props.iconContainerStyle
            ?
            this.props.iconContainerStyle
            :
            Styles.iconContainer
          }
        >
          <View
            style={
            [this.props.circleStyle
              ?
              this.props.circleStyle
              :
              Styles.circle,
            { backgroundColor: this.props.circleColor }]
          }
          >
            <Icon
              name={this.props.iconName}
              color={this.props.iconColor}
              size={this.props.iconSize}
            />
          </View>
        </View>
        <View
          style={
            this.props.messageContainerStyle
            ?
            this.props.messageContainerStyle
            :
            Styles.messageContainer
          }
        >
          <Text
            style={
              this.props.messageStyle
              ?
              this.props.messageStyle
              :
              Styles.message
            }
          >
            {this.props.message}
          </Text>
        </View>
      </View>
    );
  }
}
 
NotificationBar.propTypes = {
  backgroundColor: PropTypes.string,
  circleColor: PropTypes.string,
  circleStyle: ViewPropTypes.style,
  containerStyle: ViewPropTypes.style,
  iconColor: PropTypes.string,
  iconContainerStyle: ViewPropTypes.style,
  iconName: PropTypes.string,
  iconSize: PropTypes.number,
  message: PropTypes.string,
  messageContainerStyle: ViewPropTypes.style,
  messageStyle: Text.propTypes.style,
};
 
NotificationBar.defaultProps = {
  backgroundColor: 'black',
  circleColor: 'blue',
  circleStyle: '',
  containerStyle: '',
  iconColor: 'white',
  iconContainerStyle: '',
  iconName: '',
  iconSize: 30,
  message: '',
  messageContainerStyle: '',
  messageStyle: '',
};
