/**
 * SplashScreen Component
 * author: soorajc
 * created@: 20/07/2018
 */
 
import React, { Component } from 'react';
import {
  View,
  Text,
  ImageBackground,
  Image,
  ViewPropTypes,
} from 'react-native';
import PropTypes from 'prop-types';
 
import Styles from './styles';
 
const SplashImage = require('../../assets/splash.png');
 
export default class SplashScreen extends Component<> {
 
  componentDidMount() {
    const interval = this.props.interval;
    setTimeout(() => this.props.transitionHandler(), interval);
  }
 
  render() {
    return (
        this.props.backgroundType === 'image'
        ?
          <ImageBackground
            resizeMode="stretch"
            source={this.props.backgroundImage}
            style={
              this.props.imageBackgroundStyle
              ?
              this.props.imageBackgroundStyle
              :
              Styles.imageBackground
            }
          >
            <Text
              style={
              this.props.mainTitleStyle
              ?
              this.props.mainTitleStyle
              :
              Styles.mainTitle
            }
            >
              {this.props.mainTitle}
            </Text>
            <Text
              style={
              this.props.subTitleStyle
              ?
              this.props.subTitleStyle
              :
              Styles.subTitle
            }
            >
              {this.props.subTitle}
            </Text>
          </ImageBackground>
        :
          <View
            style={
            [this.props.containerStyle
              ?
              this.props.containerStyle
              :
              Styles.container,
              { backgroundColor: this.props.backgroundColor },
            ]
            }
          >
            <Text
              style={
              this.props.mainTitleStyle
              ?
              this.props.mainTitleStyle
              :
              Styles.mainTitle
              }
            >
              {this.props.mainTitle}
            </Text>
            <Text
              style={
              this.props.subTitleStyle
              ?
              this.props.subTitleStyle
              :
              Styles.subTitle
              }
            >
              {this.props.subTitle}
            </Text>
          </View>
    );
  }
}
 
SplashScreen.propTypes = {
  backgroundColor: PropTypes.string,
  backgroundImage: Image.propTypes.source,
  backgroundType: PropTypes.string,
  containerStyle: ViewPropTypes.style,
  imageBackgroundStyle: Image.propTypes.style,
  interval: PropTypes.number,
  mainTitle: PropTypes.string,
  mainTitleStyle: Text.propTypes.style,
  subTitle: PropTypes.string,
  subTitleStyle: Text.propTypes.style,
  transitionHandler: PropTypes.func,
};
 
SplashScreen.defaultProps = {
  backgroundColor: 'blue',
  backgroundImage: SplashImage,
  backgroundType: 'color',
  containerStyle: '',
  imageBackgroundStyle: '',
  interval: 2000,
  mainTitle: '',
  mainTitleStyle: '',
  subTitle: '',
  subTitleStyle: '',
  transitionHandler: '',
};
