/**
 * DropDown Component
 * author: soorajc
 * created@: 31/07/2018
 */
 
import React, { Component } from 'react';
import PropTypes from 'prop-types';
import {
  Modal,
  Text,
  FlatList,
  ViewPropTypes,
  TouchableOpacity,
  View,
} from 'react-native';
 
import Icon from 'react-native-vector-icons/FontAwesome';
import Button from '../../lib/button/button';
import Styles from './styles';
 
export default class DropDown extends Component<> {
 
  constructor(props) {
    super(props);
    this.state = {
      modalVisible: false,
    };
  }
 
  setModalVisible(visible) {
    this.setState({ modalVisible: visible });
  }
 
  handleSelect(item) {
    this.setState({ modalVisible: false });
    this.props.handleSelect(item);
  }
 
  keyExtractor = (item, index) => index;
 
  render() {
    return (
      <View
        style={
          this.props.containerStyle
          ?
          this.props.containerStyle
          :
          Styles.container
        }
      >
        <Modal
          animationType="fade"
          transparent
          testID="DropdownModal"
          visible={this.state.modalVisible}
          onRequestClose={() => {
            this.setModalVisible(false);
          }}
        >
          <View
            style={
              this.props.innerContainerStyle
              ?
              this.props.innerContainerStyle
              :
              Styles.innerContainer
            }
          >
            <View
              style={
                this.props.listContainerStyle
                ?
                this.props.listContainerStyle
                :
                Styles.listContainer
              }
            >
              <FlatList
                keyExtractor={this.keyExtractor}
                data={this.props.data}
                renderItem={({ item }) =>
                  <TouchableOpacity
                    testID={item}
                    onPress={() => this.handleSelect(item)}
                    style={
                      this.props.itemStyle
                      ?
                      this.props.itemStyle
                      :
                      Styles.item
                    }
                  >
                    <Text>{item}</Text>
                  </TouchableOpacity>}
              />
            </View>
            <TouchableOpacity
              testID={'CancelButton'}
              style={
                this.props.cancelButtonStyle
                ?
                this.props.cancelButtonStyle
                :
                Styles.cancelButton
              }
              onPress={() => {
                this.setModalVisible(!this.state.modalVisible);
              }}
            >
              <Text
                style={
                  this.props.cancelButtonLabelStyle
                  ?
                  this.props.cancelButtonLabelStyle
                  :
                  Styles.cancelButtonLabel
                }
              >
                Cancel
              </Text>
            </TouchableOpacity>
          </View>
        </Modal>
        <Button
          label={this.props.label}
          style={
            this.props.dropDownButtonStyle
            ?
            this.props.dropDownButtonStyle
            :
            Styles.dropDownButton
          }
          labelStyle={
            this.props.dropDownLabelStyle
            ?
            this.props.dropDownLabelStyle
            :
            Styles.dropDownLabel
          }
          onPressHandler={() => this.setModalVisible(!this.state.modalVisible)}
        />
        <TouchableOpacity
          testID={'ArrowButton'}
          style={
            this.props.dropDownArrowStyle
            ?
            this.props.dropDownArrowStyle
            :
            Styles.dropDownArrow
          }
          onPress={() => this.setModalVisible(!this.state.modalVisible)}
        >
          <Icon
            name="caret-down"
            size={this.props.iconSize}
            color={this.props.iconColor}
          />
        </TouchableOpacity>
      </View>
    );
  }
}
 
DropDown.propTypes = {
  cancelButtonLabelStyle: Text.propTypes.style,
  cancelButtonStyle: ViewPropTypes.style,
  containerStyle: ViewPropTypes.style,
  data: PropTypes.arrayOf(PropTypes.string),
  dropDownArrowStyle: ViewPropTypes.style,
  dropDownButtonStyle: ViewPropTypes.style,
  dropDownLabelStyle: Text.propTypes.style,
  handleSelect: PropTypes.func,
  iconColor: PropTypes.string,
  iconSize: PropTypes.number,
  innerContainerStyle: ViewPropTypes.style,
  itemStyle: ViewPropTypes.style,
  label: PropTypes.string,
  listContainerStyle: ViewPropTypes.style,
};
 
DropDown.defaultProps = {
  cancelButtonLabelStyle: '',
  cancelButtonStyle: '',
  containerStyle: '',
  data: '',
  dropDownArrowStyle: '',
  dropDownButtonStyle: '',
  dropDownLabelStyle: '',
  handleSelect: null,
  iconColor: '#AFB8BB',
  iconSize: 20,
  innerContainerStyle: '',
  itemStyle: '',
  label: '',
  listContainerStyle: '',
};
