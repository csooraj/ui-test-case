import { LOGIN_URL } from './config';
 
export function userLogin(username, password) {
  return fetch(LOGIN_URL, {
    method: 'POST',
    headers: {
      'Accept': 'application/json',
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      username: username,
      password: password
    })
  }).then(function(response) {
    return response.json();
  })
.then((responseData) => {
  if (responseData) {
    return responseData;
  }
}).catch(err => {
  return err;
});
}
