/**
 * Sample Login Component
 */
 
import React, { Component } from 'react';
import {
  View,
  TextInput,
  Alert,
  ActivityIndicator,
} from 'react-native';
 
import Styles from './styles';
import Button from '../../lib/button/button';
import { userLogin } from './controller';
import { LOGIN_URL } from '../../controller/apiconfig';
import { apiCallHandler } from '../../controller/apicontroller';
 
export default class Login extends Component<{}> {
 
  constructor(props) {
    super(props);
    this.state = {
      username: '',
      password: '',
      loading: false,
    };
  }
 
  loginHandler() {
    const username = this.state.username;
    const password = this.state.password;
 
    if (username === '' || password === '') {
      Alert.alert('Enter the fields');
    } else {
      const body = JSON.stringify({
        username: username,
        password: password,
      });
 
      this.setState({ loading: true });
      // userLogin(username, password).then(
      //   data => this.loginResultHandler(data));
 
      apiCallHandler('POST', LOGIN_URL, '', body).then(
        data => this.loginResultHandler(data));
    }
  }
 
  loginResultHandler(data) {
    this.setState({ loading: false });
    if (data.message) {
      Alert.alert(data.message);
    } else {
      Alert.alert('Login Successfull');
    }
  }
 
  render() {
    return (
      <View style={Styles.container}>
        <TextInput
          style={Styles.textInput}
          underlineColorAndroid="white"
          onChangeText={text => this.setState({ username: text })}
          value={this.state.username}
          placeholder="username"
        />
        <TextInput
          style={Styles.textInput}
          underlineColorAndroid="white"
          onChangeText={text => this.setState({ password: text })}
          value={this.state.password}
          placeholder="password"
          secureTextEntry
        />
        {
          this.state.loading
          ?
            <ActivityIndicator />
          :
            <Button
              label="SIGN IN"
              style={Styles.loginButton}
              onPressHandler={() => this.loginHandler()}
            />
        }
      </View>
    );
  }
}
